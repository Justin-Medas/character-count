/**
 * @file
 * Modular Header Components.
 */
import {useEffect, useState} from 'react';
import Menu from './menus/Menu';
import logo from '../../logo.svg';
import ButtonToggle from '../../helpers/buttonToggle';

/**
 * Represents a header with accessible menu.
 */
export function Header() {
  /**
   * Get Menu Data
   * @returns {JSON | Object} of website menus
   */  
  const [menus,menusData]=useState([]);
  const getData=() => {
    fetch('https://api.jsonbin.io/b/62c2e9f25d53821c308b2a6d'
    ,{
      headers : { 
        'Content-Type': 'application/json',
        'Accept': 'application/json'
       }
    }
    )
      .then(function(response){
        return response.json();
      })
      .then(function(menus) {
        menusData(menus)
      });
  }
  
  useEffect(() => {
    getData()
  },[])

  return(
    <header className="site-header">
      <div className='site-header__sticky-container'>
        <div className='grid-container'>
          <div className='site-header__logo'>
            <img src={logo} className="App-logo" alt="logo" />
          </div>
          <ButtonToggle
            buttonClass="mobile-menu-button button-toggle"
            buttonLabel="Main Menu"
            buttonId="header-menus--button-toggle"
            buttonControls="header-menus"
            buttonActiveElement='body'
            buttonActiveClass='mobile-menu-open'
            buttonToggleType='menu'
          />           
          <div id="header-menus" aria-labelledby="header-menus--button-toggle" className="site-header__main">
            <div className="site-header__bottom">
              <Menu
                  menu={menus.main}
                  menuName="main-menu accessible-menu"
                  menuAriaLabel="Main Menu"
                  menuId="main-menu"
                />                  
            </div>
          </div>      
        </div>
      </div>
    </header>
  );
}

export default Header;
